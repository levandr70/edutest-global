rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // CELTA Trainer photos
    match /celtaTrainers/{trainerId}/{allPaths=**} {
      // Allow public read access for trainer photos
      allow read: if true;
      
      // Allow write access only for authenticated users (admin)
      allow write: if request.auth != null;
    }
    
    // CELTA Resources documents
    match /celtaResources/{resourceId}/{allPaths=**} {
      // Allow public read access for resource files
      allow read: if true;
      
      // Allow write access only for authenticated users (admin)
      allow write: if request.auth != null;
    }
  }
}
